example.rs 
use symmetrica::*;

let mut st = Store::new();
let x = st.sym("x");

// Differentiate x¬≥ + 2x
let expr = st.add(vec![
    st.pow(x, st.int(3)),
    st.mul(vec![st.int(2), x])
]);

let derivative = diff(&mut st, expr, "x");
let simplified = simplify(&mut st, derivative);

println!("{}", st.to_string(simplified));
// Output: 2 + 3 * x^2
